6.base64循环解码

import base64

def decode_base64(content):
    try:
        # 尝试进行Base64解码
        return base64.b64decode(content).decode('utf-8')
    except Exception as e:
        # 如果解码失败，返回原内容
        print(f"解码失败: {e}")
        return None

def loop_base64_decode(file_path):
    with open(file_path, 'r') as file:
        content = file.read().strip()  # 读取文件并去除空格和换行符

    # 循环进行Base64解码，直到得到一个正常的文本
    while True:
        decoded_content = decode_base64(content)
        if decoded_content is None:
            print("解码结束，得到最终内容:")
            print(content)
            break
        else:
            content = decoded_content
            print("继续解码...当前内容为:")
            print(content)
            # 假设一旦获得最终结果，不再继续解码
            if "{" in content or "}" in content:  # 可以根据实际内容设定停止条件
                print("解码结束，得到最终内容:")
                print(content)
                break

# 示例调用
loop_base64_decode('base64.txt')

7.哥斯拉解密

import base64
import gzip
import re
from urllib.parse import unquote


def unpad(s):
    return s[:-ord(s[len(s) - 1:])]


def XOR(D, K):
    result = []
    for i in range(len(D)):
         c = K[i + 1 & 15]
         if not isinstance(D[i], int):
             d = ord(D[i])
         else:
              d = D[i]
         result.append(d ^ ord(c))
    return b''.join([i.to_bytes(1, byteorder='big') for i in result])


class PHP_XOR_BASE64:
    def __init__(self, pass_, key):
        self.pass_ = pass_
        self.key = key


    def decrypt_req_payload(self, payload):
        payload = payload.decode().split(self.pass_ + '=')[1]
        return gzip.decompress(XOR(base64.b64decode(unquote(payload)),self.key))

    def decrypt_res_payload(self, payload):
        payload = payload[16:-16]
        return gzip.decompress(XOR(base64.b64decode(payload.decode()),self.key))

13.二维码碎片拼接

from PIL import Image
import os
import glob


def merge_images(image_folder, output_file, rows=10, cols=10):
    # 获取所有图片文件并按文件名排序
    image_files = sorted(glob.glob(os.path.join(image_folder, '*.*')),
                         key=lambda x: os.path.basename(x).split('.')[0])

    # 确保图片数量正确
    assert len(image_files) == rows * cols, f"需要{rows * cols}张图片，当前找到{len(image_files)}张"

    # 打开第一张图片获取尺寸
    with Image.open(image_files[0]) as img:
        img_width, img_height = img.size

    # 创建新画布
    canvas = Image.new('RGB', (cols * img_width, rows * img_height))

    # 逐个粘贴图片
    for index, image_path in enumerate(image_files):
        row = index // cols  # 计算行位置
        col = index % cols  # 计算列位置

        with Image.open(image_path) as img:
            # 确保图片尺寸一致（可选：img = img.resize((img_width, img_height))）
            if img.size != (img_width, img_height):
                raise ValueError(f"图片{os.path.basename(image_path)}尺寸不一致")

            # 计算粘贴位置
            position = (col * img_width, row * img_height)
            canvas.paste(img, position)

    # 保存结果
    canvas.save(output_file)
    print(f"拼接完成！结果已保存至：{output_file}")

14.CRC32

import zipfile
import binascii
import string

def FileRead(zipname):
    try:
        f = open(zipname)         #打开⽬标⽂件
        f.close()
    except FileNotFoundError:
        print("未找到同⽬录下的压缩包⽂件" + zipname) #如果未找到⽂件，输出错误
        return                         #退出线程，进⾏详细报错
    except PermissionError:
        print("⽆法读取⽬标压缩包（⽆权限访问）")       #如果发现目标文件无权限，输出错误
        return                                 #退出线程，进行详细报错

def FourByte(zipname):
    zip_url = "./" + zipname
    file_zip = zipfile.ZipFile(zip_url)  #⽤zipfile读取指定的压缩包⽂件
    name_list = file_zip.namelist()  #使⽤⼀个列表，获取并存储压缩包内所有的⽂件名
    crc_list = []
    crc32_list = []
    print('+--------------遍历指定压缩包的CRC值 - ---------------+')
    for name in name_list:
        name_message = file_zip.getinfo(name)
        crc_list.append(name_message.CRC)
        crc32_list.append(hex(name_message.CRC))
        print('[OK] {0}: {1}'.format(name, hex(name_message.CRC)))
    print('+------------对输出的CRC值进⾏碰撞 - ----------------+')
    comment = ''
    chars = string.printable
    for crc_value in crc_list:
        for char1 in chars:
            for char2 in chars:
                for char3 in chars:
                    for char4 in chars:
                        res_char = char1 + char2 + char3 + char4  #获取任意4Byte字符
                        thicken_crc = binascii.crc32(res_char.encode())  #获取任意4Byte字符串的CRC32值
                        calc_crc = thicken_crc & 0xffffffff  #将任意4Byte字符串的CRC32值与0xffffffff进⾏与运算
                        if calc_crc == crc_value:  #匹配两个CRC32值
                           print('[Success] {}:{}'.format(hex(crc_value),res_char))
                           comment += res_char
    print('+-----------------CRC碰撞结束！！！-----------------+')
    crc32_list = str(crc32_list)
    crc32_list = crc32_list.replace('\'', '')
    print("读取成功，导出CRC列表为：" + crc32_list)                     #导出CRC列表
    if comment:
         print('CRC碰撞成功，结果为: {}'.format(comment))                  #输出CRC碰撞结果
    else:
         print('CRC碰撞没有结果，请检查压缩包内⽂件是否为4Byte！！！')

if __name__ == '__main__':

    zipname = str(input("请输⼊压缩包名字：\n4Byte >> > "))
    try:
        if zipname:
           FileRead(zipname)
           FourByte(zipname)
    except BaseException as e:
        err = str(e)
        print('脚本详细报错：' + err)
